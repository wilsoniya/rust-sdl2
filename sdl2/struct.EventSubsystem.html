Example

use sdl2::event::Event;

let sdl_context = sdl2::init().unwrap();
let event_subsystem = sdl_context.event().unwrap();

// Read up to 1024 events
let events: Vec<Event> = event_subsystem.peek_events(1024);

// Print each one
for event in events {
    println!("{:?}", event);
}
